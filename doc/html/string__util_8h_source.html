192  /*Converts an integer value into a string.
193   @param value The integer to convert.
194  @param out The character buffer to write the integer as a string into. This can be either a char* or wchar_t* buffer.
195  @param length The length of the output buffer (in character count).
196  @returns 0 on success, -1 on failure. Will fail if the buffer is either invalid
197  or not large enough to hold the converted value.*/
198  template<typename charT>
199  inline int itoa(long value, charT* out, const size_t length)
200  {
201  if (length == 0 || out == NULL)
202  { return -1; }
203  //space for a negative sign if we need it
204  const size_t signPos = value < 0 ? 1 : 0;
205  size_t i = 0;
206  if (value == 0)
207  { out[i++] = 0x30/*zero*/; }
208  else if (value < 0)
209  {
210  out[i++] = 0x2D/*minus sign*/;
211  value = -value;
212  }
213  while (value > 0 && i+1 < length)
214  {
215  out[i++] = 0x30 + value%10;
216  value /= 10;
217  }
218  /*Not enough space in the buffer or NULL terminator? Clear out the data that we copied
219  into it and return failure.*/
220  if (i+1 == length && value > 0)
221  {
222  std::memset(out,0, length*sizeof(charT));
223  return -1;
224  }
225  out[i] = 0;
226  std::reverse<charT*>(out+signPos, out+i);
227  return 0;
228  }

269  length != 1 &&

304  break;

320  {
321  //shift OR the bits into return value.
322  intValue = intValue | (digits[strPos] << (digitPos << 2));
